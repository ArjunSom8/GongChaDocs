/**
 * @file This is the main application component that defines the app's routing and structure.
 * @module App
 */

import React, { useContext } from 'react';
import GoogleTranslate from './componets/GoogleTranslate'
import { BrowserRouter, Routes, Route, useLocation } from 'react-router-dom'
import './App.css'
import NavBar from './componets/NavBar'
import ManagerView from './componets/ManagerView'
import CashierView from './componets/CashierView'
import MenuBoard from './componets/MenuBoard'
import CoffeeSeries from './pages/CoffeeSeries'
import CoffeeSeriesCashier from './pages/CoffeeSeriesCashier'
import BrewedSeries from './pages/BrewedSeries'
import BrewedSeriesCashier from './pages/BrewedSeriesCashier'
import SeasonalSeries from './pages/SeasonalSeries'
import SeasonalSeriesCashier from './pages/SeasonalSeriesCashier'
import MilkTeaSeries from './pages/MilkTeaSeries'
import MilkTeaSeriesCashier from './pages/MilkTeaSeriesCashier'
import MilkFoamSeries from './pages/MilkFoamSeries'
import MilkFoamSeriesCashier from './pages/MilkFoamSeriesCashier'
import SlushSeries from './pages/SlushSeries'
import SlushSeriesCashier from './pages/SlushSeriesCashier'
import TeaLatteSeries from './pages/TeaLatteSeries'
import TeaLatteSeriesCashier from './pages/TeaLatteSeriesCashier'
import CreativeMixSeries from './pages/CreativeMixSeries'
import CreativeMixSeriesCashier from './pages/CreativeMixSeriesCashier'
import Weather from './componets/Weather'
import ZoomProvider, { ZoomContext } from './componets/ZoomContext';
import ZoomControls from './componets/ZoomControls';


/**
 * Inner component to consume the ZoomContext and render the application content.
 *
 * @returns {JSX.Element} The rendered AppContent component.
 */
function AppContent() {
    const { zoom } = useContext(ZoomContext);

  return (
      <BrowserRouter> 
      <div style={{ zoom: `${zoom}%` }}>
          <NavBar/>
          <ZoomControls />
              <Routes>
                  <Route path="/" element={<MilkTeaSeries/>} />
                  <Route path="/milk-tea-series" element={<MilkTeaSeries/>} />
                  <Route path="/milk-tea-series-cashier" element={<MilkTeaSeriesCashier/>} />
                  <Route path="/creative-mix-series" element={<CreativeMixSeries/>} />
                  <Route path="/creative-mix-series-cashier" element={<CreativeMixSeriesCashier/>} />
                  <Route path="/milk-foam-series" element={<MilkFoamSeries/>} />
                  <Route path="/milk-foam-series-cashier" element={<MilkFoamSeriesCashier/>} />
                  <Route path="/slush-series" element={<SlushSeries/>} />
                  <Route path="/slush-series-cashier" element={<SlushSeriesCashier/>} />
                  <Route path="/tea-latte-series" element={<TeaLatteSeries/>} />
                  <Route path="/tea-latte-series-cashier" element={<TeaLatteSeriesCashier/>} />
                  <Route path="/brewed-tea-series" element={<BrewedSeries/>} />
                  <Route path="/brewed-tea-series-cashier" element={<BrewedSeriesCashier/>} />
                  <Route path="/coffee-series" element={<CoffeeSeries/>} />
                  <Route path="/coffee-series-cashier" element={<CoffeeSeriesCashier/>} />
                  <Route path="/seasonal-series" element={<SeasonalSeries/>} />
                  <Route path="/seasonal-series-cashier" element={<SeasonalSeriesCashier/>} />
                  <Route path="/manager" element={<ManagerView/>} />
                  <Route path="/cashier" element={<CashierView/>} />
                  <Route path="/menu-board" element={<MenuBoard/>}></Route>
                  <Route path="/weather" element={<Weather/>}></Route>
              </Routes>
                <GoogleTranslate/>
                </div>
      </BrowserRouter>
  )
  }

  /**
   * The main application component that wraps AppContent with ZoomProvider.
   *
   * @returns {JSX.Element} The rendered App component.
   */
  function App() {
    return (
      <ZoomProvider>
          <AppContent />
      </ZoomProvider>
    );
  }
 


export default App;
