/**
 * @file This module defines the CashierView component for the point-of-sale (POS) system.
 * @module CashierView
 */

import React, { useState } from "react";
import { Link, useLocation } from "react-router-dom";

import CheckoutTab from "./checkoutTab";
import "../styles/POSButtonCashier.css";

/**
 * Functional component representing the CashierView.
 *
 * @returns {JSX.Element} The rendered CashierView component.
 */
const CashierView = () => {
  const [selectedOption] = useState("Sales");
  const [isCheckoutPopupOpen, setIsCheckoutPopupOpen] = useState(false);
  const [menuItems] = useState([]);
  const location = useLocation();

  /**
   * Handles the click event for the "Checkout" button and opens the checkout popup.
   */
  const handleCheckoutClick = () => {
    setIsCheckoutPopupOpen(true);
  };

  /**
   * Closes the checkout popup.
   */
  const closeCheckoutPopup = () => {
    setIsCheckoutPopupOpen(false);
  };

  return (
    <div>
      <h1 className="Title">Cashier View</h1>
      <Link to="/" className="GoToHomeButton">
        Home
      </Link>
      <nav className="navbar">
        <li
          className={
            location.pathname === "/milk-tea-series-cashier" ? "active" : ""
          }
        >
          <Link to="/milk-tea-series-cashier">Milk Tea Series</Link>
        </li>
        <li
          className={
            location.pathname === "/creative-mix-series-cashier" ? "active" : ""
          }
        >
          <Link to="/creative-mix-series-cashier">Creative Mix Series</Link>
        </li>
        <li
          className={
            location.pathname === "/milk-foam-series-cashier" ? "active" : ""
          }
        >
          {" "}
          <Link to="/milk-foam-series-cashier">Milk Foam Series</Link>
        </li>
        <li
          className={
            location.pathname === "/slush-series-cashier" ? "active" : ""
          }
        >
          <Link to="/slush-series-cashier">Slush Series</Link>
        </li>
        <li
          className={
            location.pathname === "/tea-latte-series-cashier" ? "active" : ""
          }
        >
          <Link to="/tea-latte-series-cashier">Tea Latte Series</Link>
        </li>
        <li
          className={
            location.pathname === "/brewed-tea-series-cashier" ? "active" : ""
          }
        >
          <Link to="/brewed-tea-series-cashier">Brewed Tea Series</Link>
        </li>
        <li
          className={
            location.pathname === "/coffee-series-cashier" ? "active" : ""
          }
        >
          <Link to="/coffee-series-cashier">Coffee Series</Link>
        </li>
        <li
          className={
            location.pathname === "/seasonal-series-cashier" ? "active" : ""
          }
        >
          <Link to="/seasonal-series-cashier">Seasonal Series</Link>
        </li>
        <li>
          <button onClick={handleCheckoutClick}>Checkout</button>
        </li>
        {isCheckoutPopupOpen && (
          <CheckoutTab className="checkoutThing" onClose={closeCheckoutPopup} />
        )}
      </nav>
      <div className="content"></div>
    </div>
  );
};

export default CashierView;
