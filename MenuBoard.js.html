/**
 * Renders a single menu item.
 *
 * @param {Object} props - The component's props.
 * @param {Object} props.item - The menu item to render.
 * @returns {JSX.Element} The rendered menu item.
 */
function MenuItem({ item }) {
  return (
    <div className="menu-item">
      <div className="menu-item-name">{item.name}</div>
      <div className="menu-item-price">${item.price.toFixed(2)}</div>
    </div>
  );
}

/**
 * Renders a banner for a menu category.
 *
 * @param {Object} props - The component's props.
 * @param {string} props.imageUrl - The URL of the category's banner image.
 * @param {string} props.title - The title of the category.
 * @returns {JSX.Element} The rendered category banner.
 */
function CategoryBanner({ imageUrl, title }) {
  return (
    <div className="category-banner">
      <img src={imageUrl} alt={title} className="category-banner-image" />
      <h2 className="category-title">{title}</h2>
    </div>
  );
}

/**
 * Renders a menu category and its items.
 *
 * @param {Object} props - The component's props.
 * @param {Object} props.category - The menu category to render.
 * @returns {JSX.Element} The rendered menu category.
 */
function Category({ category }) {
  return (
    <div className="menu-category">
      <CategoryBanner imageUrl={category.imageUrl} title={category.title} />
      {category.items.map(item => (
        <MenuItem key={item.id} item={item} />
      ))}
    </div>
  );
}

/**
 * Renders the entire menu board with categories and items.
 *
 * @returns {JSX.Element} The rendered menu board.
 */
function MenuBoard() {
  return (
    <div className="menu-board">
      {Object.values(menuItems).map((category, index) => (
        <Category key={index} category={category} />
      ))}
    </div>
  );
}
