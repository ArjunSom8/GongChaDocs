/**
 * @file A navigation bar component for your application.
 * @module NavBar
 */

import React, { useState, useEffect } from "react";
import { Link, useLocation } from "react-router-dom";
import CheckoutTab from "./checkoutTab";

/**
 * NavBar component that provides navigation links and checkout functionality.
 *
 * @returns {JSX.Element} The rendered NavBar component.
 */
const NavBar = () => {
  const [isCheckoutPopupOpen, setIsCheckoutPopupOpen] = useState(false);
  const [hideNavBar, setHideNavBar] = useState(false);
  const location = useLocation();

  /**
   * Opens the checkout popup.
   */
  const handleCheckoutClick = () => {
    setIsCheckoutPopupOpen(true);
  };

  /**
   * Closes the checkout popup.
   */
  const closeCheckoutPopup = () => {
    setIsCheckoutPopupOpen(false);
  };

  /**
   * Handles click event for Manager View.
   * Hides the navigation bar on restricted routes.
   */
  const handleManagerClick = () => {
    setHideNavBar(true);
  };

  /**
   * Effect hook to hide the navigation bar on restricted routes.
   */
  useEffect(() => {
    const restrictedRoutes = [
      "/manager",
      "/cashier",
      "/menu-board",
      "/topping-test",
      "/milk-tea-series-cashier",
      "/creative-mix-series-cashier",
      '/milk-foam-series-cashier',
      '/slush-series-cashier',
      '/tea-latte-series-cashier',
      '/brewed-tea-series-cashier',
      '/coffee-series-cashier',
      '/seasonal-series-cashier',
      '/weather'
    ];
    setHideNavBar(restrictedRoutes.includes(location.pathname));
  }, [location]);
  return (
    !hideNavBar && (
      <nav className="navbar">
        <li
          className={location.pathname === "/milk-tea-series" ? "active" : ""}
        >
          <Link to="/milk-tea-series">Milk Tea Series</Link>
        </li>
        <li
          className={
            location.pathname === "/creative-mix-series" ? "active" : ""
          }
        >
          <Link to="/creative-mix-series">Creative Mix Series</Link>
        </li>
        <li
          className={location.pathname === "/milk-foam-series" ? "active" : ""}
        >
          {" "}
          <Link to="/milk-foam-series">Milk Foam Series</Link>
        </li>
        <li className={location.pathname === "/slush-series" ? "active" : ""}>
          <Link to="/slush-series">Slush Series</Link>
        </li>
        <li
          className={location.pathname === "/tea-latte-series" ? "active" : ""}
        >
          <Link to="/tea-latte-series">Tea Latte Series</Link>
        </li>
        <li
          className={location.pathname === "/brewed-tea-series" ? "active" : ""}
        >
          <Link to="/brewed-tea-series">Brewed Tea Series</Link>
        </li>
        <li className={location.pathname === "/coffee-series" ? "active" : ""}>
          <Link to="/coffee-series">Coffee Series</Link>
        </li>
        <li
          className={location.pathname === "/seasonal-series" ? "active" : ""}
        >
          <Link to="/seasonal-series">Seasonal Series</Link>
        </li>
        <li>
          <button onClick={handleCheckoutClick}>Checkout</button>
        </li>
        <li className={location.pathname === "/manager" ? "active" : ""}>
          <Link to="/manager" onClick={handleManagerClick}>
            Manager View
          </Link>
        </li>
        <li className={location.pathname === "/cashier" ? "active" : ""}>
          <Link to="/cashier" onClick={handleManagerClick}>
            Cashier View
          </Link>
        </li>
        <li className={location.pathname === "/menu-board" ? "active" : ""}>
          <Link to="/menu-board" onClick={handleManagerClick}>
            Menu Board
          </Link>
        </li>
        <li className={location.pathname === "/weather" ? "active" : ""}>
          <Link to="/weather" onClick={handleManagerClick}>
            Weather
          </Link>
        </li>
        {isCheckoutPopupOpen && (
          <CheckoutTab className="checkoutThing" onClose={closeCheckoutPopup} />
        )}
      </nav>
    )
  );
};

export default NavBar;
