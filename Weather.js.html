/**
 * @file This module defines a component for displaying weather information.
 * @module Weather
 */

import React, { useState } from 'react';
import '../App.css';
import '../styles/weatherStyle.css';
import { Link } from 'react-router-dom';

/**
 * A component for displaying weather information.
 *
 * @returns {JSX.Element} The rendered weather component.
 */
function Weather() {
  const [location, setlocation] = useState('College Station');
  const [weatherData, setweatherData] = useState(null);

  /**
   * Fetches weather data from the OpenWeatherMap API based on the specified location.
   */
  function getWeather() {
    const apiKey = "1e34f932e08afc0896e0cd0c8bf8577a";
    const url = `https://api.openweathermap.org/data/2.5/weather?q=${location}&appid=${apiKey}&units=metric`;
    
    fetch(url)
      .then((response) => response.json())
      .then((data) => {
        console.log(data);
        const iconCode = data.weather[0].icon;
        const iconUrl = `http://openweathermap.org/img/w/${iconCode}.png`;
        const weather = {
          location: data.name,
          temperature: `${Math.round((data.main.temp * 1.8) + 32)} Â°F`,
          humidity: `Humidity: ${data.main.humidity}%`,
          wind: `Wind: ${data.wind.speed} km/h`,
          condition: `Weather Condition: ${data.weather[0].description}`,
          icon : iconUrl,
        };
        setweatherData(weather);
      })
      .catch((error) => {
        console.error(error);
      });
  }

  return (
    <div className="WeatherFont">
      <Link to="/" className="GoToHomeButton">
        Home
      </Link>
      <div className="weatherBox">
        <div className="searchBox">
          <input type='text' placeholder='Search' value={location} onChange={(e) => setlocation(e.target.value)}/>
          <div onClick={() => { getWeather() }}>Search</div>
        </div>
          {weatherData && (
            <div className='weatherSpacing'>
              <img src={weatherData.icon} alt="Weather Icon" />
              <div className='condition'>{weatherData.condition}</div>  
              <div className='location'>{weatherData.location}</div>
              <div className='temp'>{weatherData.temperature}</div>
              <div className='humidity'>{weatherData.humidity}</div>
              <div className='wind'>{weatherData.wind}</div>
            </div> 
          )}
      </div>
    </div>
  )
}

export default Weather;
